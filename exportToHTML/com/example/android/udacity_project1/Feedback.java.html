19       @Override 
20       protected void onCreate(Bundle savedInstanceState) { 
21           final EditText eText, eText2, eText3; 
22           Button btn2; 
23    
24           super.onCreate(savedInstanceState); 
25           setContentView(R.layout.activity_feedback); 
26    
27           eText = (EditText) findViewById(R.id.name); 
28           eText2 = (EditText) findViewById(R.id.address); 
29           eText3 = (EditText) findViewById(R.id.phone); 
30           btn2 = (Button) findViewById(R.id.submit); 
31           btn2.setOnClickListener(new View.OnClickListener() { 
32               public void onClick(View v) { 
33                   String str = eText.getText().toString(); 
34                   String str2 = eText2.getText().toString(); 
35                   String str3 = eText3.getText().toString(); 
36    
37                   TelephonyManager teleMgr = (TelephonyManager) getSystemService(Feedback.TELEPHONY_SERVICE); 
38                   String localeCountry = teleMgr.getNetworkCountryIso(); 
39                   if (localeCountry != null) { 
40                       Locale loc = new Locale("",localeCountry); 
41                       localeCountry = "" + loc; 
42                   } 
43    
44                   eText.setText(""); 
45                   eText2.setText(""); 
46                   eText3.setText(""); 
47                   fillform(str, str2, str3, localeCountry); 
48               } 
49           }); 
50       } 
51    
52       private void fillform(String str, String str2, String str3, String str4) { 
53           AlertDialog.Builder alertDialogBuilder = new AlertDialog.Builder(this); 
54           if (str.length() == 0) 
55               alertDialogBuilder.setMessage("No Details to process :-("); 
56           else { 
57               alertDialogBuilder.setTitle("Your Details :"); 
58               alertDialogBuilder.setMessage("Name: " + str + "\nAddress: " + str2 + "\nPhone Number: " + str3 + "\n\nYou are accessing this app from " + str4); 
59           } 
60           AlertDialog alertDialog = alertDialogBuilder.create(); 
61           alertDialog.show(); 
62       }
